Условные операторы и циклы в JavaScript

Условные операторы и циклы помогают управлять потоком выполнения кода, делая программы гибкими и динамичными.

Оператор if...else

Оператор if выполняет блок кода, если условие истинно (true).

let age = 18;

if (age >= 18) {
    console.log("Вы совершеннолетний.");
} else {
    console.log("Вы несовершеннолетний.");
}

Оператор else if

Используется для проверки нескольких условий.

let score = 85;

if (score >= 90) {
    console.log("Отлично!");
} else if (score >= 75) {
    console.log("Хорошо!");
} else {
    console.log("Попробуйте ещё раз.");
}

Тернарный оператор

Краткая форма записи if...else.

let isMember = true;
let discount = isMember ? 10 : 0;
console.log(Ваша скидка: ${discount}%);

Оператор switch

Используется для выбора одного из нескольких возможных вариантов.

let fruit = "яблоко";

switch (fruit) {
    case "яблоко":
        console.log("Вы выбрали яблоко.");
        break;
    case "банан":
        console.log("Вы выбрали банан.");
        break;
    default:
        console.log("Неизвестный фрукт.");
}

Цикл while

Выполняет код, пока условие истинно.

let i = 0;

while (i < 5) {
    console.log(Итерация: ${i});
    i++;
}

Цикл do...while

Сначала выполняет код, затем проверяет условие.

let j = 0;

do {
    console.log(Число: ${j});
    j++;
} while (j < 5);

Цикл for

Используется для повторяющихся действий.

for (let i = 0; i < 5; i++) {
    console.log(Итерация ${i});
}

Цикл for...in

Перебирает свойства объекта.

let user = { name: "Алекс", age: 30 };

for (let key in user) {
    console.log(${key}: ${user[key]});
}

Цикл for...of

Используется для перебора массивов и других итерируемых объектов.

let numbers = [10, 20, 30];

for (let num of numbers) {
    console.log(num);
}

Операторы break и continue

break завершает цикл, а continue пропускает итерацию.

for (let i = 0; i < 10; i++) {
    if (i === 5) break; // Прекращает цикл при 5
    if (i % 2 === 0) continue; // Пропускает четные числа
    console.log(i);
}

Вложенные циклы

Один цикл внутри другого.

for (let i = 1; i <= 3; i++) {
    for (let j = 1; j <= 3; j++) {
        console.log(i=${i}, j=${j});
    }
}

Методы массивов, заменяющие циклы

Методы forEach, map, filter, reduce позволяют обходить массивы без явных циклов.

let arr = [1, 2, 3, 4, 5];

arr.forEach(num => console.log(num)); // forEach

let squared = arr.map(num => num ** 2);
console.log(squared); // map

let even = arr.filter(num => num % 2 === 0);
console.log(even); // filter

let sum = arr.reduce((acc, num) => acc + num, 0);
console.log(sum); // reduce

Оптимизация работы с циклами

Некоторые оптимизации позволяют ускорить выполнение кода.

// Кэширование длины массива
let arr = [1, 2, 3, 4, 5];
for (let i = 0, len = arr.length; i < len; i++) {
    console.log(arr[i]);
}
